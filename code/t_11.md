package main

import "fmt"

func main() {
	fmt.Println(maxArea([]int{1, 2, 3, 4, 5}))
}
func maxArea(height []int) int {
	area := 0
	for i := 0; i < len(height)-1; i++ {
		var crisis int
		for j := len(height) - 1; j > i; j-- {
			if height[j] > crisis {
				if height[j] >= height[i] {
					_area := (j - i) * height[i]
					if _area > area {
						area = _area
					}
					break
				}
				crisis = height[j]
				_area := (j - i) * crisis
				if _area > area {
					area = _area
				}
			}
		}
	}
	return area
}
